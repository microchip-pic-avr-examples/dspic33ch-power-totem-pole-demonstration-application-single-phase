62void __attribute__((__interrupt__, auto_psv)) _ADCAN0Interrupt(void)
63{
64 TP4_DPPIM_RD13_SetHigh();
65
66 ACmonitorTrigger_SetHigh();
67
68 //prestart next shared core input which is AC_N/CH15
69 //shared core Vout/CH10 input is ready because of latency from trigger to here
70 //from min. 500ns
71 ADCON3Lbits.CNVCHSEL = 15; // change ADC channel
72 ADCON3Lbits.CNVRTCH = 1; // trigger conversion
73
74 //I1
75 Phase_Values_PH1.Phase_Current.Raw = (int16_t) ADCBUF0;
76 Phase_Values_PH1.Phase_Current.Rectified = abs((int16_t) (Phase_Values_PH1.Phase_Current.Raw - Phase_Values_PH1.Phase_Current.Offset)); //phase voltage - offset=VCC/2
77
78 ACmonitorTrigger_SetLow();
79
80 if ((Phase_Values_PH1.Phase_Current.Rectified > IIN_OC) && (pwr_ctrl_state > PCS_RELAYON)) //new 06.09.2021
81 {
82 PFC_Flags.bits.OC_PH1 = 1;
83 PFC_Flags.bits.Run = 0;
84 PFC_Flags.bits.Stop = 1;
85
86 Drv_PwrCtrl_TPBLPFC_Stop();
87 }
88
89// TP4_DPPIM_RD13_SetLow();
90
91 //Vout triggered by PWM
92 Vout_Control.Vout.Raw = ADCBUF10;
93
94 if ((Vout_Control.Vout.Raw > VOUT_OV))
95 {
96 PFC_Flags.bits.OV_Vout = 1;
97 PFC_Flags.bits.Run = 0;
98 PFC_Flags.bits.Stop = 1;
99
100 Drv_PwrCtrl_TPBLPFC_Stop();
101 }
102
103// TP4_DPPIM_RD13_SetHigh();
104
105 //shared core Vout input is ready because of latency from trigger to here
106 //from min. 350ns (measured)), AD is 270ns
107 AC_N.Phase_Voltage.Raw = ADCBUF15; //AC_N
108
109 //<get AC data
110 if (MAIN_CORE_ProtocolRead(MSI1_ProtocolB, &(primary_vac_values.Data[0])))
111 {
112 dev_connect_numbers();
113 primary_vac_values.DataReceivedOK = 1;
114 }
115
116 ADCON3Lbits.CNVCHSEL = 10; // change ADC channel, triggered by TRIGA
117
118 //<< PFC handler
119 Application_handler_PH1();
120
121 IFS5bits.ADCAN0IF = 0; //clear the channel_ANA0 interrupt flag
122
123 TP4_DPPIM_RD13_SetLow();
124
125}
