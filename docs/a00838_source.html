163void __attribute__ ((weak)) RESET_CauseHandler(void)
164{
165 uint16_t resetCause = RESET_CauseGet();
166
167 if(RESET_CauseFromTrap(resetCause))
168 {
169 RESET_CauseClear(RESET_MASK_TRAPR);
170 }
171
172 if(RESET_CauseFromIllegalOpcode(resetCause))
173 {
174 RESET_CauseClear(RESET_MASK_IOPUWR);
175 }
176
177 if(RESET_CauseFromConfigurationMismatch(resetCause))
178 {
179 RESET_CauseClear(RESET_MASK_CM);
180 }
181
182 if(RESET_CauseFromExternal(resetCause))
183 {
184 RESET_CauseClear(RESET_MASK_EXTR);
185 }
186
187 if(RESET_CauseFromSoftware(resetCause))
188 {
189 RESET_CauseClear(RESET_MASK_SWR);
190 }
191
192 if(RESET_CauseFromWatchdogTimer(resetCause))
193 {
194 RESET_CauseClear(RESET_MASK_WDTO);
195 }
196}
