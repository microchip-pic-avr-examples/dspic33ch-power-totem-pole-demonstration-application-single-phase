179inline static uint16_t ADC1_ConversionResultGet( enum ADC_CHANNEL channel )
180{
181 uint16_t result = 0x0U;
182
183 switch(channel)
184 {
185 case FB_Vout:
186 result = ADCBUF10;
187 break;
188 case FB_AC_N:
189 result = ADCBUF15;
190 break;
191 case IL1_F:
192 result = ADCBUF0;
193 break;
194 case IL2_F:
195 result = ADCBUF1;
196 break;
197 default:
198 break;
199 }
200 return result;
201}

215inline static bool ADC1_IsConversionComplete(enum ADC_CHANNEL channel)
