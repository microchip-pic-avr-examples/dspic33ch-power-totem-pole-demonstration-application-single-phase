108enum MAIN_CORE_SYSTEM_STATUS MAIN_CORE_SystemStatusGet(void)
109{
110 enum MAIN_CORE_SYSTEM_STATUS systemStatus = MAIN_CORE_SYSTEM_STATUS_RUNNING_STATE;
111
112 if(SI1STATbits.MSTRST)
113 {
114 systemStatus = MAIN_CORE_SYSTEM_STATUS_IN_RESET_STATE;
115 }
116 else
117 {
118 switch(SI1STATbits.MSTPWR)
119 {
120 case 0: systemStatus = MAIN_CORE_SYSTEM_STATUS_NOT_IN_LOW_POWER_MODE;
121 break;
122 case 1: systemStatus = MAIN_CORE_SYSTEM_STATUS_IDLE_MODE;
123 break;
124 case 2: systemStatus = MAIN_CORE_SYSTEM_STATUS_SLEEP_MODE;
125 break;
126 default:systemStatus = MAIN_CORE_SYSTEM_STATUS_RUNNING_STATE;
127 break;
128 }
129 }
130 return systemStatus;
131}
