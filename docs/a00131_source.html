300enum SEC_CORE_SYSTEM_STATUS SEC_CORE1_SystemStatusGet(void)
301{
302 enum SEC_CORE_SYSTEM_STATUS systemStatus = SEC_CORE_SYSTEM_STATUS_RUNNING_STATE;
303
304 if(MSI1STATbits.SLVRST)
305 {
306 systemStatus = SEC_CORE_SYSTEM_STATUS_IN_RESET_STATE;
307 }
308 else
309 {
310 switch(MSI1STATbits.SLVPWR)
311 {
312 case 0: systemStatus = SEC_CORE_SYSTEM_STATUS_NOT_IN_LOW_POWER_MODE;
313 break;
314 case 1: systemStatus = SEC_CORE_SYSTEM_STATUS_IDLE_MODE;
315 break;
316 case 2: systemStatus = SEC_CORE_SYSTEM_STATUS_SLEEP_MODE;
317 break;
318 default:systemStatus = SEC_CORE_SYSTEM_STATUS_RUNNING_STATE;
319 break;
320 }
321 }
322 return systemStatus;
323}
