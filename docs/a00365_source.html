129// ramp speed was 1 all the time
130#define SOFT_START_RAMP_SPEED 4 //LV step is 26.9mV, HV step is 25.2mV
131#define STEP_DELAY 30 //30*100us=3ms delay between one step of SOFT_START_RAMP_SPEED
132
133PWR_CTRL_STATE_e Softstart_reference(void)
134{
135 static uint16_t SoftstartCounter = 0;
136 static PWR_CTRL_STATE_e returnvalue = 1;
137
138#ifdef VOLTAGE_LOOP
139 if (++SoftstartCounter > STEP_DELAY)
140#else
141 if (++SoftstartCounter > 20) //fixed value
142#endif
143 {
144 if (Vout_Control.Reference.Reference_Internal <= Vout_Control.Reference.Reference_Set)
145 {
146 if (((Vout_Control.Reference.Reference_Internal + SOFT_START_RAMP_SPEED) < Vout_Control.Reference.Reference_Set)
147 && ((Vout_Control.Reference.Reference_Internal + SOFT_START_RAMP_SPEED) >= SOFT_START_RAMP_SPEED))
148 {
149 Vout_Control.Reference.Reference_Internal += SOFT_START_RAMP_SPEED;
150 returnvalue = pwr_ctrl_state;
151 }
152 else
153 {
154 returnvalue = PCS_UP_AND_RUNNING;
155 Vout_Control.Reference.VOUT_dV_dt_Hystresis_High = Vout_Control.Reference.Reference_Internal + (Vout_Control.Reference.Reference_Internal >> 3);
156 Vout_Control.Reference.VOUT_dV_dt_Hystresis_Low = Vout_Control.Reference.Reference_Internal - (Vout_Control.Reference.Reference_Internal >> 4);
157 Vout_Control.Reference.VOUT_Hystresis_High = Vout_Control.Reference.Reference_Internal + (Vout_Control.Reference.Reference_Internal >> 5);
158 Vout_Control.Reference.VOUT_Hystresis_Low = Vout_Control.Reference.Reference_Internal - (Vout_Control.Reference.Reference_Internal >> 5);
159 Phase_Values_PH1.Control_Status_Flags.bits.Reference_Changed = 0;
160 }
161 }
162 else
163 {
164 if (((Vout_Control.Reference.Reference_Internal + SOFT_START_RAMP_SPEED) > Vout_Control.Reference.Reference_Set)
165 && ((Vout_Control.Reference.Reference_Internal + SOFT_START_RAMP_SPEED) >= SOFT_START_RAMP_SPEED))
166 {
167 Vout_Control.Reference.Reference_Internal -= SOFT_START_RAMP_SPEED;
168 returnvalue = pwr_ctrl_state;
169 }
170 else
171 {
172 returnvalue = PCS_UP_AND_RUNNING;
173 Vout_Control.Reference.VOUT_dV_dt_Hystresis_High = Vout_Control.Reference.Reference_Internal + (Vout_Control.Reference.Reference_Internal >> 3);
174 Vout_Control.Reference.VOUT_dV_dt_Hystresis_Low = Vout_Control.Reference.Reference_Internal - (Vout_Control.Reference.Reference_Internal >> 4);
175 Vout_Control.Reference.VOUT_Hystresis_High = Vout_Control.Reference.Reference_Internal + (Vout_Control.Reference.Reference_Internal >> 5);
176 Vout_Control.Reference.VOUT_Hystresis_Low = Vout_Control.Reference.Reference_Internal - (Vout_Control.Reference.Reference_Internal >> 5);
177 Phase_Values_PH1.Control_Status_Flags.bits.Reference_Changed = 0;
178 }
179 }
180
181 SoftstartCounter = 0;
182 }
183 else
184 returnvalue = pwr_ctrl_state;
185
186 return returnvalue;
187}
